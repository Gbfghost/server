	if( Fast_map$ == "")
	{
		next;
		mes "^33CC33【世界傳送員】^000000";
		mes "尚未有傳送紀錄";
		close;
	}
	else
	{
		next;
		.@needZeny = Fast_map_zeny;
		.@tempMFMap$ = Fast_map$;
		for(.@i=0; getd("Fast_map_Item_"+.@i)!=0; .@i++) {
			.@needItem[.@i] = getd("Fast_map_Item_"+.@i);
			.@needNums[.@i] = getd("Fast_map_Nums_"+.@i);	
		}
		
		mes "^33CC33【世界傳送員】^000000";
		mes "是否要傳送到^0000ff"+.@tempMFMap$+"^000000？";
		if(.@needZeny) 
			mes "該次傳送需花費 " + callfunc("F_InsertComma",.@needZeny) + " Zeny";
		.@arySize = getarraysize(.@needItem);
		if(.@arySize) {
			mes "該次傳送需下列物品：";
			for(.@i=0; .@i<.@arySize; .@i++)
				mes " - " + getitemname(.@needItem[.@i]) + " x " + .@needNums[.@i];
		}
		next;
		menu
		"是",Fast_yes,
		"否",Bye;
	close;
	}
	
Fast_yes:
	for(.@i=0; .@i<.@arySize; .@i++) {
		if(countitem(.@needItem[.@i]) < .@needNums[.@i])
			goto ItemError;
	}
	if(Zeny < .@needZeny)
		goto ZenyError;	
		
//    if(BaseLevel < .@needBaseLevel)
	//	goto BaseLevelError;
	
	for(.@i=0; .@i<.@arySize; .@i++)
		delitem .@needItem[.@i], .@needNums[.@i];
	set Zeny, Zeny - .@needZeny;
	
	warp Fast_map_set$,Fast_map_X,Fast_map_Y;	
	end;

			explode(.@array$,.@needNums$,",");
			for(.@j=0; .@j<.@arySize; .@j++) 
				.@needNums[.@j] = atoi(.@array$[.@j]);
		}else if(atoi(.@needItem$)) {
			.@needItem[0] = atoi(.@needItem$);
			.@needNums[0] = atoi(.@needNums$);
		}
	}
	
	mes "^33CC33【世界傳送員】^000000";
	for(.@i=0; .@i<getarraysize(.@warp_mes$); .@i++) {
		mes .@warp_mes$[.@i];

			next;

		mes "該次傳送需下列物品：";

	set	Fast_map$,.@tempMFMap$;
	set	Fast_map_X,.@WarpMapX;
	set	Fast_map_Y,.@WarpMapY;
	set Fast_map_zeny, .@needZeny;
	set Fast_map_BaseLevel, .@needBaseLevel;
	for(.@i=0; .@i<.@arySize; .@i++) {
		setd "Fast_map_Item_"+.@i, .@needItem[.@i];
		setd "Fast_map_Nums_"+.@i, .@needNums[.@i];
	}
	// 若上次的需求物品總數大於這次，則須清除。
	for(.@i=.@arySize; getd("Fast_map_Item_"+.@i)!=0; .@i++) { 
		setd "Fast_map_Item_"+.@i, 0;
		setd "Fast_map_Nums_"+.@i, 0;
	}
	warp .@WarpMap$,.@WarpMapX,.@WarpMapY;
	close;

InitAndStart:
	.@needZeny = 0;
	.@needItem$ = "";
	.@needNums$ = "";
	.@needBaseLevel$ = "";
	deletearray .@needItem[0];
	deletearray .@needNums[0];
	goto Start;
	
ZenyError:
	mes "^33CC33【世界傳送員】^000000";
	mes "傳送至 "+.@tempMFMap$+" 時…";
	mes "需要 " + callfunc("F_InsertComma",.@needZeny) + " Zeny";
	mes "你沒有足夠的金額！";
	close;
BaseLevelError:
	mes "^33CC33【世界傳送員】^000000";
	mes "傳送至 "+.@tempMFMap$+" 時…";
	mes "需要 " + callfunc("F_InsertComma",.@needBaseLevel) + " BaseLevel";
	mes "你的等級不夠！";
	close;
		
ItemError:
	mes "^33CC33【世界傳送員】^000000";
	mes "傳送至 "+.@tempMFMap$+" 時…";
	mes "需要下列物品：";
	for(.@i=0; .@i<.@arySize; .@i++)
			mes " - " + getitemname(.@needItem[.@i]) + " x " + .@needNums[.@i];
	mes "你身上沒有足夠的物品！";
	close;	

//-------------------加入我的最愛-------------------//
AddFavorite:
	Favorites_note(.@WarpMap$,.@tempMFMap$,.@WarpMapX,.@WarpMapY,.@needZeny,.@needItem$,.@needNums$,.@needBaseLevel$);
	goto InitAndStart;
	
MyFavorites:		
	next;
	
	.@menu$ = "";
	.@temp_Max = $@Favorites_Max + Favorites_size;
	for(.@i=1; .@i<=.@temp_Max; .@i++) {
		.@menu$ += "^0750b5"+getd("Favorites_"+.@i+"$")+"^000000:";
	}
	.@menu$ += "擴增選項數量:回最上層";
	
	mes "^33CC33【世界傳送員】^000000";
	mes "請問您要傳送到哪裡呢?";
	next;
	select(.@menu$);
	if(@menu == .@temp_Max + 1) {
		.@upItem = 52511;
		.@upItemName$ = getitemname(.@upItem);
		
		mes "^33CC33【世界傳送員】^000000";
		mes "如果你給我"+callfunc("F_ItemLink",.@upItem)+"的話";
		mes "我可以幫你擴增一個位子";
		mes "你想要擴增嗎？";
		next;
		if(select("確定:取消")==1) {
			if(Favorites_size >= 3) {				
				.@upItem = 52512;
				.@upItemName$ = getitemname(.@upItem);
				
				if(countitem(.@upItem) == 0) {
					mes "^33CC33【世界傳送員】^000000";
					mes "很抱歉，最多只能擴增3次!";
					close2;
					goto InitAndStart;
				}else{					
					mes "^33CC33【世界傳送員】^000000";
					mes "你身上有"+callfunc("F_ItemLink",.@upItem)+"耶!";
					mes "如果你願意把它給我的話，";
					mes "我願意再幫你擴增位子!";
					next;
					if(select("答應:拒絕") == 2) {
						close2;
						goto InitAndStart;
					}else if(Favorites_size == 9) {
						mes "^33CC33【世界傳送員】^000000";
						mes "很抱歉，最多只能擴增9次!";
						close2;
						goto InitAndStart;
					}
				}
			}
			
			if(countitem(.@upItem)) {
				delitem .@upItem, 1;
				Favorites_size++;
				.@tempIndex = $@Favorites_Max + Favorites_size;
				setd "Favorites_set_"+.@tempIndex+"$","尚未新增";
				setd "Favorites_"+.@tempIndex+"$","尚未新增";
				
				mes "^33CC33【世界傳送員】^000000";
				mes "已經成功擴增我的最愛數量!";
				close2;
				goto InitAndStart;
			}else{
				mes "^33CC33【世界傳送員】^000000";
				mes "你身上沒有"+callfunc("F_ItemLink",.@upItem)+"啊!";
				close2;
				goto InitAndStart;
			}
		}
	}else if(@menu == .@temp_Max + 2) {
		close2;
		goto InitAndStart;
	}else
		WarpFavorites(@menu);
	end;
	
ErrorChoose:
	mes "^33CC33【世界傳送員】^000000";
	mes "尚未新增地圖";
	next;
	goto InitAndStart;

	next;

	mes "您是否要將我的最愛列表重置刪除?";
	next;

		goto InitAndStart;

			setd "Favorites_Nums_"+.@select+"_"+.@i,0;	
		}

	}

	function	WarpFavorites	{
		disable_items;
		.@num = getarg(0);
		.@needZeny = getd("Favorites_Zeny_"+.@num);
		if( getd("Favorites_"+.@num+"$") == "尚未新增" )
			goto ErrorChoose;
		if(Zeny < .@needZeny) {
			mes "^33CC33【世界傳送員】^000000";
			mes "該次傳送需求 "+callfunc("F_InsertComma",.@needZeny)+" Zeny";
			mes "你沒有足夠的金額！";
			close;
		}
		
		for(.@i=1; getd("Favorites_Item_"+.@num+"_"+.@i)!=0; .@i++) {
			.@needItem = getd("Favorites_Item_"+.@num+"_"+.@i);
			.@needNums = getd("Favorites_Nums_"+.@num+"_"+.@i);
			if(countitem(.@needItem) < .@needNums) {
				mes "^33CC33【世界傳送員】^000000";
				mes "該次傳送需求物品清單中的…";
				mes " - " + getitemname(.@needItem) + " x " + .@needNums + " 個";
				mes "你身上沒有足夠的物品！";
				close;
			}
		}
		
		for(.@i=1; getd("Favorites_Item_"+.@num+"_"+.@i)!=0; .@i++) {
			.@needItem = getd("Favorites_Item_"+.@num+"_"+.@i);
			.@needNums = getd("Favorites_Nums_"+.@num+"_"+.@i);
			delitem .@needItem, .@needNums;
		}
		set Zeny, Zeny - .@needZeny;
		warp getd("Favorites_set_"+.@num+"$"),getd("Favorites_X_"+.@num),getd("Favorites_Y_"+.@num);
		end;
	}

	function	Favorites_note	{

		.@isReapeat = 0;
		.@temp_Max = $@Favorites_Max + Favorites_size;
		for(.@i=1; .@i<=.@temp_Max; .@i++) {
			if( getd("Favorites_"+.@i+"$") == getarg(1) ) {		
				.@isReapeat = 1;
				break;
			}		
		}
		
		if(.@isReapeat == 1) {
			next;
			mes "^33CC33【世界傳送員】^000000";
			mes "重複登錄我的最愛，已存在列表中";
			next;
			return 0;
		}
		
		.@isFull = 1;
		for(.@i=1; .@i<=.@temp_Max; .@i++) {
			if( getd("Favorites_"+.@i+"$") == "尚未新增" )
			{
				setd "Favorites_set_"+.@i+"$",getarg(0);
				setd "Favorites_"+.@i+"$",getarg(1);
				setd "Favorites_X_"+.@i,getarg(2);
				setd "Favorites_Y_"+.@i,getarg(3);
				if(getargcount()>=5) setd "Favorites_Zeny_"+.@i, getarg(4);
				if(getargcount()==7) {					
					explode(.@array$,getarg(5),",");
					.@arySize = getarraysize(.@array$);
					if(.@arySize != 0) {					
						for(.@j=0; .@j<.@arySize; .@j++) 
							.@needItem[.@j] = atoi(.@array$[.@j]);
						explode(.@array$,getarg(6),",");
						for(.@j=0; .@j<.@arySize; .@j++) 
							.@needNums[.@j] = atoi(.@array$[.@j]);

						for(.@j=0; .@j<.@arySize; .@j++) {							
							setd "Favorites_Item_"+.@i+"_"+(.@j+1), .@needItem[.@j];
							setd "Favorites_Nums_"+.@i+"_"+(.@j+1), .@needNums[.@j];
						}
					}else if(atoi(getarg(5))) {
						setd "Favorites_Item_"+.@i+"_1", atoi(getarg(5));
						setd "Favorites_Nums_"+.@i+"_1", atoi(getarg(6));
					}
				}
				.@isFull = 0;
				break;
			}
		}
		
		if(.@isFull == 1){
			next;
			mes "^33CC33【世界傳送員】^000000";
			mes "我的最愛已滿，請移除後在重新新增。";
			next;
			return 0;
		}
		next;
		mes "^33CC33【世界傳送員】^000000";
		mes "已經儲存完成";
		next;
		return 0;
	} 
	end;
