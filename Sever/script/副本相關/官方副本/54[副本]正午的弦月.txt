rachel,176,145,6	script	夢境通道#poent	PORTAL,{
	mes "[夢境通道]";
	mes "你好，這裡是夢境通道，你需要甚麼呢？";
	next;
	switch(select("漫步於夢境中", "取消")) {
	case 1:
			.@party_id = getcharid(1);
			.@p_name$ = getpartyname(.@party_id); 
			.@md_name$ = "正午的弦月";

			if (!instance_check_party(.@party_id,1)) {
				mes "[夢境通道]";
				mes "你需要一個隊伍才能進入。";
				close;
			}
			if (getcharid(0) != getpartyleader(.@party_id,2)) {
				mes "[夢境通道]";
				mes "你需要獨自一人參加才可以。";
				close;
			}
			
			switch(instance_enter(.@md_name$)) {
			case IE_OTHER:
				mes "[尼樂研大神官]";
				mes "啊…現在不是合適的時間，";
				mes "請你晚點再來。";
				close;
			case IE_NOINSTANCE:
			case IE_NOMEMBER:
				mes "[尼樂研大神官]";
				mes "你還不能進入夢境…";
				close;
			case IE_OK:
				set getvariableofinstance('party_id, instance_id(IM_PARTY)), getcharid(1);
				announce "★ 副本公告 ★ 隊伍 [ "+getpartyname(getcharid(1)) +" ] 準備挑戰【 正午的弦月 】副本！",15,0x33ea91;
				if (checkquest(14586) == -1) setquest 14586;	
				//warp "1@pop1",47,133;
				close;
			}
	case 2:
		close;
	}
}
